let map, panelOpen = false; function togglePanel() { panelOpen = !panelOpen; document.getElementById('panel').classList.toggle('hidden', !panelOpen); document.getElementById('toggle-label').textContent = panelOpen ? 'Hide Intel' : 'View Intel'; setTimeout(function() { google.maps.event.trigger(map, 'resize'); }, 350); } function showError(msg) { document.getElementById('error').style.display = 'flex'; document.getElementById('error-msg').textContent = msg; } function formatDesc(desc) { if (!desc) return []; return desc.split('|').map(function(s) { return s.trim(); }).filter(function(s) { return s.length > 2; }); } function toggleSection(header) { const body = header.nextElementSibling; const chevron = header.querySelector('.chevron'); body.classList.toggle('collapsed'); chevron.style.transform = body.classList.contains('collapsed') ? 'rotate(-90deg)' : ''; } function buildPanel(d) { document.getElementById('panel-title').textContent = d.name || 'Property'; document.getElementById('panel-address').textContent = d.address || ''; document.getElementById('property-name').textContent = d.name || 'Property'; const views = [ { label: 'Alpha', badge: 'A', cls: '', data: d.alpha }, { label: 'Bravo', badge: 'B', cls: '', data: d.bravo }, { label: 'Charlie', badge: 'C', cls: '', data: d.charlie }, { label: 'Delta', badge: 'D', cls: '', data: d.delta }, { label: 'Nadir', badge: 'N', cls: 'nadir', data: d.nadir } ]; let html = ''; views.forEach(function(v) { if (!v.data || !v.data.url) return; const bullets = formatDesc(v.data.desc); html += `
${v.badge}
${v.label}
▼
    ${bullets.map(function(b) { return '
  • ' + b + '
    • '; }).join('')}
`; }); if (d.viewer360) { html += `
360
360° Aerial View
▼
  • Full interactive drone flyover of the property
  • Pan, tilt, and zoom to explore every angle
  • Complete site overview before arrival
↗ Launch 360° Viewer
`; } document.getElementById('panel-content').innerHTML = html; } function initMap(lat, lng) { map = new google.maps.Map(document.getElementById('map'), { center: { lat: lat, lng: lng }, zoom: 18, mapTypeId: google.maps.MapTypeId.SATELLITE, tilt: 0, mapTypeControl: true, mapTypeControlOptions: { style: google.maps.MapTypeControlStyle.DROPDOWN_MENU, mapTypeIds: ['satellite', 'hybrid', 'roadmap'] }, zoomControl: true, streetViewControl: true, fullscreenControl: true }); const marker = new google.maps.Marker({ position: { lat: lat, lng: lng }, map: map, animation: google.maps.Animation.DROP }); marker.addListener('click', function() { if (!panelOpen) togglePanel(); }); setTimeout(function() { if (!panelOpen) togglePanel(); }, 600); } window.initGoogleMaps = async function() { const id = new URLSearchParams(window.location.search).get('id'); if (!id) { showError('No property ID provided.'); return; } try { const res = await fetch('https://jonahbourgeois1.github.io/property-intel/data/' + id + '.json'); if (!res.ok) { showError('Property "' + id + '" not found.'); return; } const d = await res.json(); buildPanel(d); initMap(d.lat, d.lng); } catch(e) { showError('Failed to load property data: ' + e.message); } };
